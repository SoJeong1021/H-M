create table commentdb (
comment_num int(11) auto_increment primary key,
name varchar(100) ,
content varchar(1000), 
board_num varchar(45) ,
wdate timestamp ,
userID varchar(45));

create table follow(
userID varchar(45) ,
follow_user varchar(45)
);


create table imgg(
id int(11) auto_increment primary key,
FILENAME varchar(50) ,
imgpath varchar(45) ,
imgCate varchar(45)
);

create table songg(
id int(11) auto_increment primary key,
FILENAME varchar(100) ,
songpath varchar(100) ,
songCate varchar(45));


create table pair(
id int(11) auto_increment primary key,
userid varchar(45) ,
imgid int(11) ,
imgName varchar(45) ,
imgDesc varchar(1000) ,
songid int(11) ,
songName varchar(45) ,
songDesc varchar(1000));

create table user (
id varchar(45) primary key,
passwd varchar(45) ,
name varchar(45) ,
phone_number varchar(45) ,
nickname varchar(45) ,
gender varchar(45)
);
